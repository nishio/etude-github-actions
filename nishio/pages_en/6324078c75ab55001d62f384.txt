code:py
 import numpy as np
 for dim in [2, 10, 100, 200, 300, 400]:
     over020 = 0
     over022 = 0
     over024 = 0
     for i in range(10000):
         x = np.random.randn(dim)
         x = x / np.linalg.norm(x)
         if x[0] > 0.2:
             over020 += 1
         if x[0] > 0.22:
             over022 += 1
         if x[0] > 0.24:
             over024 += 1
 
     print(dim, over020, over022, over024)
