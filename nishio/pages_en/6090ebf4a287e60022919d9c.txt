 class RGB(object):
     def __init__(self, rgb):
         self.rgb = map(float, rgb)
 
     @staticmethod
     def s(s):
         "build from #ffffff form"
         if s[0] == "#": s = s[1:]
         v = int(s, 16)
         r = (v & 0xFF0000) >> 16
         g = (v & 0xFF00) >> 8
         b = v & 0xFF
         return RGB([r, g, b])
 
     def hsv(self):
         rgb = self.rgb
         r, g, b = rgb
         MAX = max(rgb)
         MIN = min(rgb)
         DIFF = MAX - MIN
         h = {
             r: 60 * (g - b) / DIFF,
             g: 60 * (b - r) / DIFF + 120,
             b: 60 * (r - g) / DIFF + 240,
         }[max(rgb)]
         s = DIFF / MAX
         v = MAX
         return HSV([h, s, v])
 
     def __str__(self):
         "round to web color"
         return "#%s" % ("".join(
             "0123456789ABCDEF"[i] * 2
             for i in [int(round(x / 17))
                       for x in self.rgb]))
                         
 class HSV(object):
     def __init__(self, hsv):
         self.hsv = map(float, hsv)
 
     def rotate(self, h):
         self.hsv[0] = (self.hsv[0] + h) % 360
 
     def darken(self, k):
         from math import log, exp
         v = self.hsv[2]
         v = min(log(exp(v) * k), 255)
         self.hsv[2] = v
         
 
     def rgb(self):
         h, s, v = self.hsv
         hi, f = divmod(h, 60)
         f /= 60
         p = v * (1 - s)
         q = v * (1 - s * f)
         t = v * (1 - s * (1 - f))
         return RGB([
             [v, t, p],
             [q, v, p],
             [p, v, t],
             [p, q, v],
             [t, p, v],
             [v, p ,q],
         ][int(hi)])
